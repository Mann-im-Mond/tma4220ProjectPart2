\section{ODE }

\begin{frame}
\frametitle{ODE in Matrix form}
	\begin{block}{}
		\begin{flalign*}
			M\dot{u}=Au+v(t)
		\end{flalign*}
	\end{block}
	\begin{block}{}
		\begin{flalign*}
			\dot{u}=M^{-1}(Au+v(t)).
		\end{flalign*}
	\end{block}
\end{frame}

\begin{frame}
\frametitle{Explicit solvers}
	\begin{block}{}
		\begin{itemize}
			\item Forward Euler
			\begin{flalign*}
				u_{n+1}=u_n+hM^{-1}(Au_n+v(t_n))
			\end{flalign*}
			\item Improved Forward Euler
			\begin{flalign*}
				u_{n+1}=u_n+\frac{h}{2}M^{-1}&\left(2Au_n+v(t_n)+v(t_{n+1}) \right. \\
					&\left.+hA\cdot M^{-1}\left(Au_n+v(t_n)\right)\right)
			\end{flalign*}
		\end{itemize}
	\end{block}
\end{frame}

\begin{frame}
\frametitle{Implicit solvers}
	\begin{block}{}
		\begin{itemize}
			\item Backwards Euler
			\begin{flalign*}
				(M-hA)u_{n+1}=Mu_n+hv(t_n)
			\end{flalign*}
			\item Crank-Nicholson
			\begin{flalign*}
				(M-\frac{h}{2}A)u_{n+1}=(M+\frac{h}{2}A)u_n+\frac{h}{2}(v(t_n)+v(t_{n+1}))
			\end{flalign*}
		\end{itemize}
	\end{block}
\end{frame}
